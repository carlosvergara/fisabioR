
%--------------------------------------------------------%
% La clase fisabior es la empleada en los informes       %
% estadísticos realizados con el paquete de R fisabior.  %
% Esta clase se basa en el paquete memoir, y puede       %
% customizarse para otras salidas                        %
%--------------------------------------------------------%

\documentclass[$layout$]{fisabior}

$if(biblio-files)$
    \addbibresource{$biblio-files$}
$endif$

$if(numbersections)$
    \setcounter{secnumdepth}{5}
$else$
    \setcounter{secnumdepth}{0}
$endif$

\title{$title$}

$for(author)$
  $if(author.affil.number)$
    \author[$author.affil.number$]{
  $else$
    \author{
  $endif$
      $author.name$
      $if(author.corresponding)$
        \corremail{$author.corresponding$}
        \thanks{\corresponding}
      $endif$
    }
$endfor$

$if(affil)$
  $for(affil)$
    $if(affil.number)$
      \affil[$affil.number$]{$affil.name$}
    $else$
      \affil{$affil.name$}
    $endif$
  $endfor$
$endif$


$if(version)$
  \version{$version$}
$endif$

$if(bibliography)$
  \addbibresource{$bibliography.file$}
$endif$

%-------------------------------------------%
% Inicio del documento                      %
%-------------------------------------------%

\begin{document}

\logotitulo{$logo$}
\maketitle

$if(abstract)$
  \begin{abstract}
    $abstract$
  \end{abstract}
$endif$

$body$

$if(bibliography)$
  $if(bibliography.title)$
    \printbibliography[title = $bibliography.title$]
  $else$
    \printbibliography[title = Referencias Bibliográficas]
  $endif$
$endif$

\end{document}
